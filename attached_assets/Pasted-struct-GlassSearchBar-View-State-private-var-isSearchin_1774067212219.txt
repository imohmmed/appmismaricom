struct GlassSearchBar: View {
    @State private var isSearching = false
    @State private var searchText = ""
    @FocusState private var isFocused: Bool // للتحكم بالكيبورد

    var body: some View {
        HStack {
            if isSearching {
                // شكل حقل البحث عند التمدد
                HStack {
                    Image(systemName: "magnifyingglass")
                        .foregroundColor(.gray)
                    
                    TextField("Search", text: $searchText)
                        .focused($isFocused)
                        .foregroundColor(.white)
                    
                    if !searchText.isEmpty {
                        Button(action: { searchText = "" }) {
                            Image(systemName: "xmark.circle.fill")
                                .foregroundColor(.gray)
                        }
                    }
                }
                .padding(.horizontal, 10)
                .frame(maxWidth: .infinity) // يتمدد ليأخذ كل المساحة
                .frame(height: 40)
                .background(.ultraThinMaterial)
                .cornerRadius(20)
                
                // زر الإلغاء (الـ X الكبير اللي بالصورة)
                Button(action: {
                    withAnimation(.spring()) {
                        isSearching = false
                        searchText = ""
                        isFocused = false
                    }
                }) {
                    Image(systemName: "xmark")
                        .padding(10)
                        .background(.ultraThinMaterial)
                        .clipShape(Circle())
                        .foregroundColor(.white)
                }
            } else {
                // الشكل الأصلي (الأقسام + الزر الصغير)
                HStack {
                    Text("All").padding(.horizontal, 8)
                    Text("GC").padding(.horizontal, 8)
                    Text("DM").padding(.horizontal, 8)
                }
                .padding(10)
                .background(.ultraThinMaterial)
                .cornerRadius(20)
                
                Spacer()
                
                // زر البحث الصغير
                Button(action: {
                    withAnimation(.spring()) {
                        isSearching = true
                        isFocused = true // يفتح الكيبورد تلقائياً
                    }
                }) {
                    Image(systemName: "magnifyingglass")
                        .padding(10)
                        .background(.ultraThinMaterial)
                        .clipShape(Circle())
                        .foregroundColor(.white)
                }
            }
        }
        .padding()
    }
}
