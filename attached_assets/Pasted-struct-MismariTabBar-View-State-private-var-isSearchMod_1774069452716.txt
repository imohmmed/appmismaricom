struct MismariTabBar: View {
    @State private var isSearchMode = false
    @State private var searchText = ""
    
    var body: some View {
        HStack(spacing: 12) {
            
            // --- الحاوية الأولى: الأقسام ---
            HStack {
                if isSearchMode {
                    // عندما نبحث: يظهر فقط زر الرجوع للرئيسية
                    Button(action: { withAnimation(.spring()) { isSearchMode = false } }) {
                        Image(systemName: "house.fill")
                            .padding(10)
                    }
                } else {
                    // الحالة الطبيعية: كل الأقسام
                    HStack(spacing: 15) {
                        TabItem(icon: "plus.square.fill", label: "PLUS+")
                        TabItem(icon: "play.tv", label: "TV")
                        TabItem(icon: "message", label: "SMM")
                    }
                    .padding(.horizontal, 10)
                }
            }
            .frame(height: 50)
            .background(.ultraThinMaterial)
            .clipShape(Capsule())
            
            // --- الحاوية الثانية: البحث ---
            HStack {
                if isSearchMode {
                    // شريط البحث الممتد
                    HStack {
                        Image(systemName: "magnifyingglass")
                        TextField("Search Mismari...", text: $searchText)
                            .foregroundColor(.primary)
                    }
                    .padding(.horizontal, 15)
                } else {
                    // زر البحث الدائري الصغير
                    Button(action: { withAnimation(.spring()) { isSearchMode = true } }) {
                        Image(systemName: "magnifyingglass")
                            .font(.system(size: 20, weight: .bold))
                            .frame(width: 50, height: 50)
                    }
                }
            }
            .frame(maxWidth: isSearchMode ? .infinity : 50) // هنا سحر التمدد والتقلص
            .frame(height: 50)
            .background(.ultraThinMaterial)
            .clipShape(Capsule())
            
        }
        .padding(.horizontal)
        .padding(.bottom, 20)
    }
}
